79  template<typename Class, typename>
80  void deserialize(Class& obj, const json& object)
81  {
82  if (object.is_object()) {
83  reflect::for_all_members<Class>([&obj, &object](auto& member) {
84  auto iter = object.find(std::string(member.get_name()));
85  if (iter == object.end()) throw util::exception("Error: Member '{}' not found in json", member.get_name());
86  auto& objName = *iter;
87  if (!objName.is_null()) {
88  using MemberT = reflect::get_member_type<decltype(member)>;
89  if constexpr (std::decay_t<decltype(member)>::can_get_ref()) {
90  deserialize(member.get_ref(obj), objName);
91  } else if (member.has_setter()) {
92  if constexpr (std::is_default_constructible_v<MemberT> &&
93  std::is_assignable_v<MemberT&, MemberT&&>) {
94  MemberT m{};
95  deserialize(m, objName);
96  member.set(obj, std::move(m));
97  } else {
98  throw util::exception("Error: can't deserialize member '{}' because a reference can "
99  "not be accessed and the type is not default constructible",
100  member.get_name());
101  }
102  } else {
103  throw util::exception("Error: can't deserialize member '{}' because it's read only",
104  member.get_name());
105  }
106  }
107  });
108  } else {
109  throw util::exception("Error: can't deserialize from Json::json to {}. Json: {}", reflect::get_name<Class>(), object);
110  }
111  }
