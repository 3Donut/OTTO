202  template<typename C>
203  inline void Waveform<C>::PointCache::refresh()
204  {
205  float new_spp = wf._range.size() / wf.size.w;
206  float ratio = smpl_pr_px / new_spp;
207  float new_radius = radius * ratio;
208  if (new_radius > wf.radius_range.out) {
209  radius = wf.radius_range.in;
210  } else if (new_radius < wf.radius_range.in) {
211  radius = wf.radius_range.out;
212  } else {
213  radius = new_radius;
214  }
215  smpl_pr_px = new_spp;
216 
217  // Refresh the cache, but keep allocated memory
218  points.clear();
219 
220  if (wf._range.size() == 0 || wf.container.size() == 0) {
221  return;
222  }
223 
224  sp2r = 2 * radius * smpl_pr_px;
225  double error = 0.f;
226  auto first = std::begin(wf.container);
227  auto fidx = clamp(std::floor(wf._range.in / sp2r) * sp2r);
228  auto lidx = clamp(std::ceil(wf._range.out / sp2r) * sp2r);
229  auto iter = first + fidx;
230  auto last = first + lidx;
231  float start_x = (int(fidx) - int(wf._range.in)) / smpl_pr_px;
232  for (int i = 0; iter != last; i++, iter++) {
233  double intprt = 0;
234  error = std::modf(error + sp2r, &intprt);
235 
236  // Average:
237  float sum = 0.f;
238  for (int j = 0; (j < intprt) && (iter != last); j++, iter++) {
239  sum += std::abs(*iter);
240  }
241  float val = sum / intprt;
242  // Max
243  // float val = 0;
244  // for (int j = 0; (j < intprt) && (iter != last); j++, iter++) {
245  // val = std::max(std::abs(*iter), val);
246  // }
247 
248  points.emplace_back(
249  start_x + i * 2 * radius,
250  (1 - std::min(val / wf._top_val, 1.f)) * wf.size.h);
251  // Required, as iter might otherwise be incremented twice with no check
252  if (iter == last) break;
253  }
254 
255  // Align points if they are too close
256  for (auto [cur, nxt] : util::adjacent_pairs(points)) {
257  float dx = std::abs(nxt.x - cur.x);
258  float dy = std::abs(nxt.y - cur.y);
259  if (dx < radius || dy < radius) {
260  nxt.y = cur.y;
261  }
262  }
263 
264  // Move first and last points to edges
265  if (!points.empty()) {
266  points.front().x = 0.f;
267  points.back().x = wf.size.w;
268  }
269  }
