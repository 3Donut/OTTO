299  AudioBufferHandle allocate()
300  {
301  for (std::size_t i = 0; i < reference_counts.size(); i++) {
302  if (reference_counts[i] < 1) {
303  if (i > _max_val) {
304  LOGI("Using {} buffers", i + 1);
305  _max_val = i;
306  }
307  reference_counts[i] = 0;
308  int index = i * buffer_size;
309  return {data.get() + index, buffer_size, reference_counts[i]};
310  }
311  }
312  DLOGI("Allocating a new audio buffer. this pool now has {} buffers", _avaliable_buffers);
313  reserve(_avaliable_buffers + 1);
314  return allocate();
315  }

338  void reserve(std::size_t n)
339  {
340  if (_avaliable_buffers > 0) {
341  LOGF("Requested {} buffers. only {} avaliable", n, _avaliable_buffers);
342  return;
343  }
344  if (n > _avaliable_buffers) {
345  data = std::make_unique<float[]>(n * buffer_size);
346  _avaliable_buffers = n;
347  reference_counts.resize(_avaliable_buffers, 0);
348  }
349  }
