185  AudioBufferHandle allocate() noexcept
186  {
187  for (std::size_t i = 0; i < reference_counts.size(); i++) {
188  if (reference_counts[i] < 1) {
189  if (i > _max_val) {
190  LOGI("Using {} buffers", i + 1);
191  _max_val = i;
192  }
193  reference_counts[i] = 0;
194  int index = i * buffer_size;
195  return {data.get() + index, buffer_size, reference_counts[i]};
196  }
197  }
198  // TODO: handle this reasonably
199  LOGF("No free audio buffers found. This shouldn't happen");
200  std::terminate();
201  }
