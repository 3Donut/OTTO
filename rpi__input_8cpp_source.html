33  static int open_device(const std::string& device_type, bool block = false)
34  {
35  auto path = fs::path("/dev/input/by-id");
36 
37  for (const auto& entry : fs::directory_iterator(path)) {
38  auto file = entry.path().string();
39  if (ends_with(file, device_type)) {
40  auto fullpath = path / file;
41  auto fd = open(fullpath.c_str(), O_RDONLY | (block ? 0 : O_NONBLOCK));
42  LOGI("Opening device {}", fullpath);
43  if (fd < 0) {
44  LOGE("Couldn't open a file descriptor for {}", fullpath.string());
45  return -1;
46  }
47 
48  auto result = ioctl(fd, EVIOCGRAB, 1);
49  if (result != 0) {
50  LOGE("Couldn't get exclusive input access to {}", fullpath.string());
51  return -1;
52  }
53 
54  return fd;
55  }
56  }
57 
58  return -1;
59  }
