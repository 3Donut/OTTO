66  audio::ProcessData<0> Euclid::process(audio::ProcessData<0> data)
67  {
68  auto& current = current_channel();
69  // Copy for thread safety
70  if (auto recording = this->recording; recording) {
71  for (auto& event : data.midi) {
72  util::match(event,
73  [&](midi::NoteOnEvent& ev) {
74  if (!_has_pressed_keys) {
75  util::fill(recording.value(), -1);
76  current.notes.set(recording.value());
77  _has_pressed_keys = true;
78  }
79  for (auto& note : recording.value()) {
80  if (note >= 0) continue;
81  note = ev.key;
82  break;
83  }
84  util::unique(recording.value(), std::equal_to<char>());
85  current.notes = recording.value();
86  },
87  [&](midi::NoteOffEvent& ev) {
88  for (auto& note : recording.value()) {
89  if (note != ev.key) continue;
90  note = -1;
91  }
92  if (util::all_of(recording.value(), [](int note) { return note < 0; })) {
93  recording = std::nullopt;
94  }
95  },
96  [](auto&&) {});
97  this->recording = recording;
98  if (!recording) break;
99  ;
100  }
101  }
102  if (_should_run) running = true;
103  if (!running) return data;
104 
105  if (!_should_run && running) {
106  _counter = _samples_per_beat;
107  running = false;
108  return data;
109  }
110 
111  auto next_beat = _samples_per_beat - _counter;
112 
113  if (next_beat <= data.nframes) {
114  for (auto& channel : props.channels) {
115  if (channel.length > 0) {
116  channel._beat_counter++;
117  channel._beat_counter %= channel.length;
118  if (running && channel._hits_enabled.at(channel._beat_counter)) {
119  for (auto note : channel.notes.get()) {
120  if (note >= 0) {
121  data.midi.push_back(midi::NoteOnEvent(note));
122  data.midi.push_back(midi::NoteOffEvent(note));
123  }
124  }
125  }
126  }
127  }
128  }
129 
130  _counter += data.nframes;
131  _counter %= _samples_per_beat;
132  return data;
133  }

301  void EuclidScreen::refresh_state()
302  {
303  auto& props = engine.props;
304  auto& current = props.channels.at(props.channel);
305 
306  state.max_length = 0;
307  for (auto& chan : props.channels) {
308  if (state.max_length < chan.length) state.max_length = chan.length;
309  }
310 
311  for (int i = 0; i < 4; i++) {
312  auto& chan = props.channels.at(i);
313  auto& cs = state.channels.at(i);
314 
315  cs.channel = &chan;
316 
317  cs.radius = 40 + i * 20;
318  cs.is_current = &chan == &current;
319  cs.length = chan.length;
320 
321  for (int i = 0; i < cs.length; i++) {
322  auto& hs = cs.hits.at(i);
323  float angle = 2.0 * M_PI * (i / float(state.max_length) - 0.25);
324  hs.point = state.center + Point{cs.radius * std::cos(angle), cs.radius * std::sin(angle)};
325  hs.active = chan._hits_enabled.at(i);
326  }
327  }
