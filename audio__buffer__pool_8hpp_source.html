163  AudioBufferHandle allocate() noexcept
164  {
165  for (std::size_t i = 0; i < reference_counts.size(); i++) {
166  if (reference_counts[i] < 1) {
167  if (i > _max_val) {
168  LOGI("Using {} buffers", i + 1);
169  _max_val = i;
170  }
171  reference_counts[i] = 0;
172  int index = i * buffer_size;
173  return {data.get() + index, buffer_size, reference_counts[i]};
174  }
175  }
176  // TODO: handle this reasonably
177  LOGF("No free audio buffers found. This shouldn't happen");
178  std::terminate();
179  }
