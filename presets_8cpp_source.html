93  void load_preset_files()
94  {
95 
96  LOG_SCOPE_FUNCTION(INFO);
97  if (!fs::exists(presets_dir)) {
98  DLOGI("Creating preset directory");
99  fs::create_directories(presets_dir);
100  }
101  DLOGI("Loading presets");
102  for (auto&& de : fs::recursive_directory_iterator(presets_dir)) {
103  if (de.is_directory()) continue;
104  auto filename = de.path().filename().string();
105  if (filename.size() == 0 || filename.c_str()[0] == '.') continue;
106  LOGI_SCOPE("Loading preset file {}", de.path());
107  if (de.is_regular_file() || de.is_symlink()) {
108  util::JsonFile jf{de.path()};
109  jf.read();
110  std::string engine = jf.data()["engine"];
111  std::string name = jf.data()["name"];
112  auto& pd = _preset_data[engine];
113  if (auto found = util::find(pd.names, name); found != pd.names.end()) {
114  pd.data[found - pd.names.begin()] = std::move(jf.data()["props"]);
115  DLOGI("Reloaded preset '{}' for engine '{}", name, engine);
116  } else {
117  pd.names.push_back(name);
118  pd.data.emplace_back(std::move(jf.data()["props"]));
119  DLOGI("Loaded preset '{}' for engine '{}", name, engine);
120  }
121  }
122  }
