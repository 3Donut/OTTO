13  /// A sum type, holding either an Ok value, or an Err
14  ///
15  /// Modelled closely after Rust's Result enum.
16  ///
17  /// # Example
18  /// ```cpp
19  /// result<std::string, int> open_file(File& f) {
20  /// f.open();
21  /// if (f.error()) {
22  /// return f.errorCode();

25  /// }
