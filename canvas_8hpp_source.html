454  template<typename It>
455  Canvas& plotBezier(It pointB, It pointE, float f = 0.5, float t = 1) {
456 
457  moveTo(*pointB);
458 
459  float m = 0;
460  Point d1;
461 
462  Point prev = *pointB;
463  Point cur = prev;
464  Point next = prev;
465  for (auto it = pointB; it != pointE; ++it) {
466  prev = cur;
467  cur = next;
468  next = *it;
469  if (cur == prev) continue;
470  m = (next.y - prev.y) / (next.x - prev.x);
471  Point d2 = {
472  (next.x - cur.x) * -f,
473  0
474  };
475  d2.y = d2.x * m * t;
476  bezierCurveTo(prev - d1, cur + d2, cur);
477  d1 = d2;
478  }
479  return *this;
480  }

482  /// Draw a curve between points using only horizontal and vertical lines and
483  /// rounded corners
484  ///
485  /// This is the algorithm used by the samplers, and in general to display waveforms
486  ///
487  /// \param first An iterator to the first point
488  /// \param last An iterator one past the last point
489  /// \param maxR The maximum radius for curves, in pixels. If negative, there
490  /// is no limit
491  /// \requires `It` shall be an `InputIterator<Point>`
492  template<typename Iter>
493  void plotRounded(Iter first, Iter last, float max_radius)
494  {
495  if (first == last) return;
496 
497  moveTo(*first);
498  for (auto [p1, p2] : util::adjacent_pairs(first, last))
499  {
500  float dx = std::abs(p2.x - p1.x);
501  float dy = std::abs(p2.y - p1.y);
502  float r = std::min({dx / 2.f, dy / 2.f, max_radius});
503  vg::Point md = (p1 + p2) / 2.0;
504  vg::Point cp1 = {md.x, p1.y};
505  vg::Point cp2 = md;
506  vg::Point cp3 = {md.x, p2.y};
507 
508  arcTo(cp1, cp2, r);
509  arcTo(cp3, p2, r);
510  }
511  lineTo(*(--last));
512  }
