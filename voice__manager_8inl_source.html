173  template<typename V, int N>
174  auto VoiceManager<V, N>::get_voice(int key) noexcept -> Voice&
175  {
176  if (free_voices.size() > 0) {
177  auto it = util::find_if(note_stack, [key](NoteVoicePair& nvp) { return nvp.note == key; });
178  auto fvit = it == note_stack.end() ? free_voices.begin() : util::find(free_voices, it->voice);
179  auto& v = **fvit;
180  free_voices.erase(fvit);
181  return v;
182  } else {
183  auto found = util::find_if(note_stack, [](NoteVoicePair& nvp) { return nvp.has_voice(); });
184  if (found != note_stack.end()) {
185  DLOGI("Stealing voice {} from key {}", (found->voice - voices_.data()), found->note);
186  Voice& v = *found->voice;
187  v.release();
188  found->voice = nullptr;
189  return v;
190  } else {
191  DLOGE("No voice found. Using voice 0");
192  return voices_[0];
193  }
194  }

197  template<typename V, int N>
198  auto VoiceManager<V, N>::stop_voice(int key) noexcept -> Voice*
199  {
200  auto free_voice = [this](Voice& v) {
201  auto found = std::find_if(note_stack.rbegin(), note_stack.rend(),
202  [](auto nvp) { return !nvp.has_voice(); });
203  if (found != note_stack.rend()) {
204  found->voice = &v;
205  // TODO: Restore original velocity
206  v.trigger(found->note, v.velocity_);
207  } else {
208  v.release();
209  free_voices.push_back(&v);
210  }
211  };
212  Voice* res = nullptr;
213  auto it = util::remove_if(note_stack, [&](auto nvp) {
214  if (nvp.note == key) {
215  if (nvp.has_voice()) {
216  free_voice(*nvp.voice);
217  res = nvp.voice;
218  }
219  return true;
220  }
221  return false;
222  });
223  note_stack.erase(it, note_stack.end());
224  return res;
225  }
