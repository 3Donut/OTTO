143  template<typename V, int N>
144  auto VoiceManager<V, N>::IVoiceAllocator::get_voice(int key, int note) noexcept -> Voice&
145  {
146  if (vm.free_voices.size() > 0) {
148  auto fvit = vm.free_voices.begin();
150  auto it = util::find_if(vm.voices_, [note](Voice& vp) { return vp.midi_note_ == note; });
152  if (it != vm.voices_.end()) {
154  if (!it->is_triggered()) fvit = util::find(vm.free_voices, it);
156  }
157  auto& v = **fvit;
158  vm.free_voices.erase(fvit);
159  return v;
160  } else {
161  auto found = util::find_if(vm.note_stack, [](NoteVoicePair& nvp) { return nvp.has_voice(); });
162  if (found != vm.note_stack.end()) {
163  DLOGI("Stealing voice {} from key {}", (found->voice - vm.voices_.data()), found->note);
164  Voice& v = *found->voice;
165  v.release();
166  found->voice = nullptr;
167  return v;
168  } else {
169  DLOGE("No voice found. Using voice 0");
170  return vm.voices_[0];
171  }

175  template<typename V, int N>
176  void VoiceManager<V, N>::IVoiceAllocator::stop_voice(int key) noexcept {
177  auto free_voice = [this](Voice& v) {
178  //TODO: this used to be reversed. Does it cause problems in poly?
179  auto found = std::find_if(vm.note_stack.begin(), vm.note_stack.end(),
180  [](auto nvp) { return !nvp.has_voice(); });
181  if (found != vm.note_stack.end()) {
182  found->voice = &v;
183  v.trigger(found->note, found->detune, found->velocity, vm.settings_props.legato, false);
184  } else {
185  v.release();
186  vm.free_voices.push_back(&v);
187  }
188  };
189 
190  auto it = std::remove_if(vm.note_stack.begin(), vm.note_stack.end(), [&](auto nvp) {
191  if (nvp.key == key) {
192  if (nvp.has_voice()) {
193  free_voice(*nvp.voice);
194  }
195  return true;
196  }
197  return false;
198  });
199  vm.note_stack.erase(it, vm.note_stack.end());
200  }

251  template<typename V, int N>
252  void VoiceManager<V, N>::MonoAllocator::handle_midi_on(const midi::NoteOnEvent& evt) noexcept
253  {
254  constexpr int num_voices_used = 3;
255  auto& vm = this->vm;
256  auto key = evt.key + services::UIManager::current().state.octave * 12 + vm.settings_props.transpose;
257  this->stop_voice(key);
258  if (vm.note_stack.size() > 0) {
259  for (int i = 0; i < num_voices_used; ++i) {
260  int sv = i>0; //Are we dispatching a subvoice?
261  // Every iteration in the loop, a new entry is added to the notestack
262  auto& note = *(vm.note_stack.begin() + num_voices_used - 1);
263  DLOGI("Stealing voice {} from key {}", (note.voice - vm.voices_.data()), note.note);
264  Voice &v = *note.voice;
267  if (!vm.settings_props.legato) v.release_no_env();
268  note.voice = nullptr;
269  vm.note_stack.push_front({.key = key, .note = key - 12 * i, .detune = 1, .velocity = evt.velocity / 127.f, .voice = &v});
270  v.trigger(key - 12 * sv, 1, evt.velocity * (1 - sv + vm.settings_props.sub * (float)sv) / 127.f, vm.settings_props.legato, false);
271  }
272  }
273  else {
274  for (int i = 0; i < num_voices_used; ++i) {
275  int sv = i>0; //Are we dispatching a subvoice?
276  auto fvit = vm.free_voices.begin() + i;
277  auto& v = **fvit;
278  vm.note_stack.push_front({.key = key, .note = key - 12 * sv, .detune = 1, .velocity = evt.velocity / 127.f, .voice = &v});
279  v.trigger(key - 12 * sv, 1, evt.velocity * (1 - sv + vm.settings_props.sub * (float)sv) / 127.f, false, vm.settings_props.retrig);
280  }
281  }

336  template<typename V, int N>
337  VoiceManager<V, N>::VoiceManager(Props& props) noexcept : props(props)
338  {
339 
340  for (int i = 0; i < voice_count_v; ++i) {
341  auto& voice = voices_[i];
342  envelope_props.attack.on_change().connect(
343  [&voice](float attack) { voice.env_.attack(8 * attack * attack + 0.02); });
344  envelope_props.decay.on_change().connect([&voice](float decay) { voice.env_.decay(decay + 0.02); });
345  envelope_props.sustain.on_change().connect(
346  [&voice](float sustain) { voice.env_.sustain(sustain); });
347  envelope_props.release.on_change().connect(
348  [&voice](float release) { voice.env_.release(4 * release * release + 0.02); });
349 
350  settings_props.portamento.on_change()
351  .connect([&voice](float p) {
352  voice.glide_ = voice.glide_.getEnd();
353  voice.glide_.period(p);
354  }).call_now(settings_props.portamento);
355  }
356 
357  settings_props.detune.on_change().connect([this](float det){
358  detune_values.clear();
359  detune_values.push_back(1);
360  for (int i = 1; i<3; i++) {
361  detune_values.push_back(1 + det * 0.015f * i);
362  detune_values.push_back(1.f / (1.f + det * 0.015f * (float)i));
363  }
364  for (auto&& [v, d] : util::zip(voices_, detune_values)) v.glide_ = midi::note_freq(v.midi_note_) * d;
365  }).call_now(settings_props.detune);
366 
367  settings_props.rand.on_change().connect([this](float r){
368  rand_values.clear();
369  for (int i = 0; i<voice_count_v; i++) {
370  rand_values.push_back(rand_max[i] * r - r + 1.f);
371  }
372  }).call_now(settings_props.rand);
373 
374  // The second and third voice are sub voices on mono mode. This sets their volume.
375  for (int i = 1; i < 3; ++i) {
376  auto& voice = voices_[i];
377  settings_props.sub.on_change().connect([&voice](float s){
378  voice.env_.amp(s);
379  });
380  }
381 
382  settings_props.play_mode.on_change()
383  .connect([this](PlayMode mode) {
384  switch (mode) {
385  case PlayMode::poly: voice_allocator = std::make_unique<PolyAllocator>(*this); break;
386  case PlayMode::mono: voice_allocator = std::make_unique<MonoAllocator>(*this); break;
387  case PlayMode::unison: voice_allocator = std::make_unique<UnisonAllocator>(*this); break;
388  case PlayMode::interval: voice_allocator = std::make_unique<IntervalAllocator>(*this); break;
389  default:break;
390  }
391  })
392  .call_now(settings_props.play_mode);
393 
394  sustain_.on_change().connect([this](bool val) {
395  if (!val) {
396  auto copy = note_stack;
397  for (auto&& nvp : copy) {
398  if (nvp.should_release) {
399  voice_allocator->stop_voice(nvp.note);
400  DLOGI("Released note {}", nvp.note);
401  }
402  }
403  }
404  });
405  }
