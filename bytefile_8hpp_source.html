256  result<void, OutIter> ByteFile::read_bytes(OutIter f, OutIter l) {
257  if (!is_open()) throw Error(Error::Type::FileNotOpen);
258  result<void, OutIter> res {std::monostate()};
259  // If OutIter is a pointer, copy everything at once
260  if constexpr (std::is_pointer_v<OutIter>) {
261  fstream.read((char*) f, l - f);
262  if (fstream.eof()) {
263  res = f + fstream.gcount() / sizeof(std::remove_pointer_t<OutIter>);
264  fstream.clear();
265  }
266  } else {
267  char* ptr;
268  std::streamsize i = 0;
269  for (OutIter iter = f; iter != l; iter++, i++) {
270  fstream.read(ptr, 1);
271  if (fstream.eof()) {
272  res = iter;
273  fstream.clear();
274  break;
275  }

277  }
278  }
