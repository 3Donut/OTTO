5 /// OTTO Metaprogramming utilities
6 ///
7 /// ## Metafunctions
8 /// A metafunction is a struct that takes template arguments, and its "return"
9 /// value is either a `static constexpr const` member variable named `value` or
10 /// a memeber type named `type`. These can be accessed using the
11 /// `meta::_t<Func>` or `meta::_v<Func>` shorthand metavariables.
12 ///
13 /// ## Metavariables
14 /// Metavariables are either constexpr variables, or alias declarations, and
15 /// should be prefixed by an underscore, to indicate that they do not need to be
16 /// unwrapped using `_t` or `_v` like the metafunctions
17 ///
18 /// ## Metatypes
19 /// The center of most metafunctions here is the `list` metatype. This is
20 /// simply a list of types that can be manipulated using metafunctions like
21 /// `sort`, `uniquify`, `flatten` etc.
22 /// There is also `pair` and `one`, which are useful mainly internally in
23 /// metafunctions
24 ///
25 /// For more information, take a look at the test file
26 namespace otto::meta {

400  template<typename Pivot, template<class T1, class T2> typename Compare, typename... Types>
401  struct partition<list<Types...>, Pivot, Compare> {
402  private:
403  constexpr static auto impl()
404  {
405  if constexpr (sizeof...(Types) == 0) {
406  return pair<list<>, list<>>();
407  } else {
408  using List = list<Types...>;
409  using H = _t<head<List>>;
410  using R = _t<partition<_t<tail<List>>, Pivot, Compare>>;
411  if constexpr (_v<Compare<H, Pivot>>) {
412  return pair<_t<concat<list<H>, _first<R>>>, _second<R>>();
413  } else {
414  return pair<_first<R>, _t<concat<list<H>, _second<R>>>>();
415  }
416  }
417  }
418  public:
419  using type = decltype(impl());
420  };
