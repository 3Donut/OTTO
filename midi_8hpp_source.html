184  inline AnyMidiEvent from_bytes(gsl::span<unsigned char> bytes, int time = 0)
185  {
186  if (bytes.size() < 3) throw util::exception("Midi event size must be >= 3 bytes");
187  auto type = MidiEvent::Type(bytes[0] >> 4);
188  auto velocity = bytes[2];
189  switch (type) {
190  case MidiEvent::Type::NoteOff: return NoteOffEvent::from_bytes(bytes, time);
191  case MidiEvent::Type::NoteOn: {
193  if (velocity != 0)
194  return NoteOnEvent::from_bytes(bytes, time);
195  else
196  return NoteOffEvent::from_bytes(bytes, time);
197  }
198  case MidiEvent::Type::ControlChange: return ControlChangeEvent::from_bytes(bytes, time);
199  case MidiEvent::Type::PitchBend: return PitchBendEvent::from_bytes(bytes, time);
200  default: return MidiEvent::from_bytes(bytes, time);
201  }
202  }
