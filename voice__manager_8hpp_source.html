172  template<int N>
173  auto VoiceManager<N>::get_voice(char key) -> Voice
174  {
175  if (free_voices.size() > 0) {
176  auto it = util::find_if(note_stack, [key] (auto& nvp) { return nvp.note == key; });
177  auto fvit = it == note_stack.end() ? free_voices.begin() : util::find(free_voices, it->voice);
178  auto v = *fvit;
179  free_voices.erase(fvit);
180  return v;
181  } else {
182  auto found =
183  util::find_if(note_stack, [](auto& nvp) { return nvp.has_voice(); });
184  if (found != note_stack.end()) {
185  Voice v = found->voice;
186  found->voice = Voice{-1};
187  voices[v].midi.trigger = false;
188  DLOGI("Stealing voice {} from key {}", v, found->note);
189  return v;
190  } else {
191  DLOGE("No voice found. Using voice 0");
192  return Voice{0};
193  }
194  }
195  }

197  template<int N>
198  void VoiceManager<N>::stop_voice(char key)
199  {
200  auto free_voice = [this](Voice v) {
201  auto found = std::find_if(note_stack.rbegin(), note_stack.rend(),
202  [](auto nvp) { return !nvp.has_voice(); });
203  if (found != note_stack.rend()) {
204  found->voice = v;
205  auto& vp = voices[v];
206  vp.midi.trigger = true;
207  vp.midi.freq = midi::note_freq(found->note + settings_props.octave * 12 + settings_props.transpose);
208  } else {
209  auto& vp = voices[v];
210  free_voices.push_back(v);
211  vp.midi.trigger = false;
212  }
213  };
214  auto it = util::remove_if(note_stack, [key, &free_voice](auto nvp) {
215  if (nvp.note == key) {
216  if (nvp.has_voice()) free_voice(nvp.voice);
217  return true;
218  }
219  return false;
220  });
221  note_stack.erase(it, note_stack.end());
222  }
