243  void JackAudioDriver::gatherMidiInput(int nframes)
244  {
245  midi_bufs.swap();
246 
247  // Get new midi events
248  void* midiBuf = jack_port_get_buffer(ports.midiIn, nframes);
249  int nevents = jack_midi_get_event_count(midiBuf);
250 
251  jack_midi_event_t event;
252  for (int i = 0; i < nevents; i++) {
253  using namespace core::midi;
254 
255  jack_midi_event_get(&event, midiBuf, i);
256  MidiEvent mEvent;
257 
258  mEvent.channel = event.buffer[0] & 0b00001111;
259  std::copy_n(event.buffer + 1, std::min(event.size - 1, mEvent.max_data_size),
260  mEvent.data.begin());
261  mEvent.time = event.time;
262 
263  auto type = MidiEvent::Type(event.buffer[0] >> 4);
264  switch (type) {
265  case MidiEvent::Type::NoteOff:
266  mEvent.type = MidiEvent::Type::NoteOff;
267  midi_bufs.inner().emplace_back(NoteOffEvent(mEvent));
268  break;
269  case MidiEvent::Type::NoteOn:
270  mEvent.type = MidiEvent::Type::NoteOn;
271  midi_bufs.inner().emplace_back(NoteOnEvent(mEvent));
272  break;
273  case MidiEvent::Type::ControlChange:
274  mEvent.type = MidiEvent::Type::ControlChange;
275  midi_bufs.inner().emplace_back(ControlChangeEvent(mEvent));
276  break;
277  }
278  }
