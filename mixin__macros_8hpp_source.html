157 /// Start a mixin declaration.
158 ///
159 /// First argument is the mixin tag name, the rest (maximum 14) are other mixins
160 /// to inherit from.
161 ///
162 /// Usage:
163 /// ```cpp
164 /// OTTO_PROPS_MIXIN(has_maximum, has_value)
165 /// {
166 /// OTTO_PROPS_MIXIN_DECLS(has_maximum);
167 /// static_assert(util::is_number_v<T>, "property mixin 'has_maximum' requires
168 /// a numeric type");
169 /// T maximum() const;
170 /// };
171 /// ```
172 /// This defines the tag type `has_name`, and the type template
173 /// `has_name_impl<T>`, which inherits virtually from
174 /// [otto::props::PropertyBase<T>]() and `has_value`. Applying this mixin will
175 /// make the member function `maximum` avaliable to the user, along with any
176 /// `has_value` members.
177 #define OTTO_PROPS_MIXIN(...)
178  /* Call OTTO_PROPS_MIXIN_1 if only 1 argument was provided, and */
179  /* OTTO_PROPS_MIXIN_2 if multiple args were provided */
180  SWITCH_FOR_VARARGS_1(OTTO_PROPS_MIXIN_1, OTTO_PROPS_MIXIN_2, __VA_ARGS__)
