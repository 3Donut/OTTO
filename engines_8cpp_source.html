127  {
128  // Main processor function

130  auto seq_out = sequencer->process(midi_in);
131  auto audio_out = service::audio::buffer_pool().allocate_multi<2>();
132  auto synth_out = synth->process({external_in.audio, seq_out.midi, external_in.nframes});
133 
134  for (auto&& [snth, out] : util::zip(synth_out, audio_out)) {
135  std::get<0>(out) = std::get<0>(snth);
136  std::get<1>(out) = std::get<0>(snth);
137  }
138  synth_out.audio.release();
139  auto fx_out = effect->process(seq_out.redirect(audio_out));
140  audio_out.release();
141 
142  return master.process(std::move(fx_out));
143  }
