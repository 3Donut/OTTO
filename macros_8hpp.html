inline constexpr Enum operator & (Enum lhs, Enum rhs) \
{ \
return Enum{std::underlying_type_t<Enum>(lhs) & \
std::underlying_type_t<Enum>(rhs)}; \
} \
inline constexpr Enum operator | (Enum lhs, Enum rhs) \
{ \
return Enum{std::underlying_type_t<Enum>(lhs) | \
std::underlying_type_t<Enum>(rhs)}; \
} \
inline constexpr Enum operator &= (Enum lhs, Enum rhs) \
{ \
return lhs = (lhs & rhs); \
} \
inline constexpr Enum operator |= (Enum lhs, Enum rhs) \
{ \
return lhs = (lhs | rhs); \
}
