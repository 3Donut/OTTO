280 inline Tv Env_lite<N,Tv,Tp,Td>::operator()(){
281 
282  // Sustain stage:
283  if(sustained()){
284  return mLevels[mStage] * voice_();
285  }
286 
287  // Interpolating along segment:
288  else if(mPos < mLen){
289  ++mPos;
290  return mCurve() * voice_();
291  }
292 
293  // Just went past end of current segment and there are more left:
294  else if(mStage < size()){
295  ++mStage;
296  if(mLoop && done()) mStage=0;
297  if(!done()){
298  mPos = 0;
299  setLen(mStage);
300  int nextStage = mStage+1;
301  // If looping, ensure we wrap back around to first level
302  if(mLoop && (nextStage==size())) nextStage = 0;
303  mCurve.set(mLen, mCurves[mStage], mLevels[mStage], mLevels[nextStage]);
304 
305  // Immediately return start level of new stage
306  return (*this)() * voice_();
307  }
308  }
309 
310  // Envelope is done. No call to parent voice here:
311  return mLevels[mStage];
312 }
