121  static void setup_alsa_device(snd_pcm_t* handle, int channels)
122  {
123  int rc;
124  snd_pcm_hw_params_t* params;
125 
126  /* Allocate a hardware parameters object. */
127  rc = snd_pcm_hw_params_malloc(&params);
128  LOGF_IF(rc < 0, "Unable to allocate hardware param struct: {}",
129  snd_strerror(rc));
130 
131  /* Fill it in with default values. */
132  rc = snd_pcm_hw_params_any(handle, params);
133  LOGF_IF(rc < 0, "Unable to initialize hardware param struct: {}",
134  snd_strerror(rc));
135 
136  /* Set the desired hardware parameters. */
137 
138  /* Interleaved mode */
139  rc = snd_pcm_hw_params_set_access(handle, params,
140  SND_PCM_ACCESS_RW_INTERLEAVED);
141  LOGE_IF(rc < 0, "Error setting access: {}", snd_strerror(rc));
142 
143  /* Signed 16-bit little-endian format */
144  rc = snd_pcm_hw_params_set_format(handle, params, SND_PCM_FORMAT_FLOAT);
145  LOGE_IF(rc < 0, "Error setting format: {}", snd_strerror(rc));
146 
147  /* Two channels (stereo) */
148  rc = snd_pcm_hw_params_set_channels(handle, params, channels);
149  LOGE_IF(rc < 0, "Error setting channelse: {}", snd_strerror(rc));
150 
151  unsigned rate = AudioDriver::get().samplerate;
152  int dir = 0;
153  rc = snd_pcm_hw_params_set_rate_near(handle, params, &rate, &dir);
154  LOGE_IF(rc < 0, "Error setting sample rate: {}", snd_strerror(rc));
155  AudioDriver::get().samplerate = rate;
156 
157  // Set periods
158  rc = snd_pcm_hw_params_set_periods_near(handle, params, &periods, 0);
159  LOGE_IF(rc < 0, "Error setting periods: {}", snd_strerror(rc));
160 
161  snd_pcm_uframes_t frame_size = channels * sizeof(float);
162  snd_pcm_uframes_t frames = buffer_size * periods / frame_size;
163  rc = snd_pcm_hw_params_set_buffer_size_near(handle, params, &frames);
164  LOGE_IF(rc < 0, "Error setting buffer size: {}", snd_strerror(rc));
165 
166  if (frames != buffer_size * periods / frame_size) {
167  buffer_size = frames * frame_size / periods;
168  }
169 
170  /* Display information about the PCM interface */
171 
172  LOGI("PCM handle name = '{}'\n", snd_pcm_name(handle));
173 
174  LOGI("PCM state = {}\n", snd_pcm_state_name(snd_pcm_state(handle)));
175 
176  debug_snd_params(params);
177 
178  /* Write the parameters to the driver */
179  rc = snd_pcm_hw_params(handle, params);
180  LOGF_IF(rc < 0, "Unable to set HW params: {}", snd_strerror(rc));
181  }
