915  template<typename It>
916  Canvas& Canvas::plotBezier(It pointB, It pointE, float f, float t)
917  {
918  moveTo(*pointB);
919 
920  float m = 0;
921  Point d1;
922 
923  Point prev = *pointB;
924  Point cur = prev;
925  Point next = prev;
926  for (auto it = pointB; it != pointE; ++it) {
927  prev = cur;
928  cur = next;
929  next = *it;
930  if (cur == prev) continue;
931  m = (next.y - prev.y) / (next.x - prev.x);
932  Point d2 = {(next.x - cur.x) * -f, 0};
933  d2.y = d2.x * m * t;
934  bezierCurveTo(prev - d1, cur + d2, cur);
935  d1 = d2;
936  }
937  return *this;
938  }

940  template<typename Iter>
941  Canvas& Canvas::plotRounded(Iter first, Iter last, float max_radius)
942  {
943  if (first == last) return *this;
944 
945  moveTo(*first);
946  auto prev = first;
947  first++;
948  for (auto iter = first; iter != last; iter++, prev++) {
949  auto p1 = *prev;
950  auto p2 = *iter;
951  float dx = std::abs(p2.x - p1.x);
952  float dy = std::abs(p2.y - p1.y);
953  float r = std::min({dx / 2.f, dy / 2.f, max_radius});
954  Point md = (p1 + p2) / 2.0;
955  Point cp1 = {md.x, p1.y};
956  Point cp2 = md;
957  Point cp3 = {md.x, p2.y};
958 
959  arcTo(cp1, cp2, r);
960  arcTo(cp3, p2, r);
961  }
962  lineTo(*(--last));
963  return *this;
964  }
